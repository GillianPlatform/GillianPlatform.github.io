type tt =
  1. | Skip
  2. | VarAssign of string * WExpr.t
    (*

    x := e

    *)
  3. | Fresh of string
    (*

    fresh x

    *)
  4. | New of string * int
    (*

    x := new(k)

    *)
  5. | Dispose of WExpr.t
    (*

    free(e)

    *)
  6. | Lookup of string * WExpr.t
    (*

    x := e

    *)
  7. | Update of WExpr.t * WExpr.t
    (*

    e := e

    *)
  8. | FunCall of string * string * WExpr.t list * (string * (string * WLExpr.t) list) option
    (*

    x := f(e1, ..., en), last bit should be ignored

    *)
  9. | While of WExpr.t * t list
    (*

    while (e) { s }

    *)
  10. | If of WExpr.t * t list * t list
    (*

    if (e) { s } else { s }

    *)
  11. | Par of t list
  12. | Logic of WLCmd.t
    (*

    logic command

    *)
  13. | Assert of WExpr.t
    (*

    non-SL assertion

    *)
  14. | Assume of WExpr.t
    (*

    non-SL assumption

    *)
  15. | AssumeType of WExpr.t * WType.t
    (*

    type assumption

    *)
